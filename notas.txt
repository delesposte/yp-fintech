#Ferramenta para auxilio durante o Debug (TSC -> Node JS -> Nodemon)
npm i --save-dev ts-node-dev
npx ts-node-dev .\src\index.ts

flags:
--respawn Reinicia sem deixar objeto na memória
--transpile-only Transpila ignorando erros de tipagem
--ignore-watch node_modules Ignora as alterações na pasta node_modules 
--no-notify Não notifica quando der um restart na aplicação
--rs Permite restart com rs
--cls lLimpa a tela ao reiniciar

Adicionar script
  "debug": "npx tsnd --respawn --transpile-only --ignore-watch node_modules --no-notify --cls --rs src/index.ts",

Testar a ferramenta
  npx run debug
